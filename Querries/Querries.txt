PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT ?book
WHERE {
  ?book rdf:type bookrec:Book .
  ?book bookrec:suitableForReadingLevel bookrec:Beginner .
}

PREFIX bookrec: <http://example.org/bookrec#>

SELECT ?book ?user
WHERE {
  ?user rdf:type bookrec:User .
  ?user bookrec:hasPreferredTheme bookrec:ScienceFiction .
  ?user bookrec:hasReadingLevel bookrec:Intermediate .
  ?book rdf:type bookrec:Book .
  ?book bookrec:hasTheme bookrec:ScienceFiction .
  ?book bookrec:suitableForReadingLevel bookrec:Intermediate .
}

  bookrec:Alice bookrec:hasReadingLevel ?readingLevel .
